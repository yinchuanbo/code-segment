视频实时录制，并设置镜像翻转及圆角，下载视频样式同设置

<button id="startButton">开始</button>

* {

  margin: 0;
}
html,
body {
  width: 100%;
  height: 100%;

  display: flex;
  flex-direction: column;
  align-items: center;

  width: 200px;
  height: 40px;
  margin: 10px 0;

const startButton = document.getElementById("startButton");
const stopButton = document.getElementById("stopButton");
const videoElement = document.getElementById("videoElement");
let mediaRecorder;
let recordedChunks = [];

startButton.onclick = async () => {
  const videoElement = document.createElement("video");
  videoElement.autoplay = true;
  videoElement.style.display = "none";

  const canvasElement = document.createElement("canvas");
  const canvasContext = canvasElement.getContext("2d");

  document.body.appendChild(videoElement);
  document.body.appendChild(canvasElement);

  let mediaRecorder;
  let recordedChunks = [];

  const audioStream = await navigator.mediaDevices.getUserMedia({
    audio: true,
  });
  const videoStream = await navigator.mediaDevices.getUserMedia({
    video: true,
  });

  videoElement.srcObject = videoStream;

  videoElement.onloadedmetadata = () => {
    canvasElement.width = videoElement.videoWidth;
    canvasElement.height = videoElement.videoHeight;
  };

  const startRecording = () => {
    recordedChunks = [];
    const canvasStream = canvasElement.captureStream();
    const mergedStream = new MediaStream([
      ...audioStream.getTracks(),
      ...canvasStream.getTracks(),
    ]);
    mediaRecorder = new MediaRecorder(mergedStream);
    mediaRecorder.ondataavailable = (event) => {
      recordedChunks.push(event.data);
    };
    mediaRecorder.onstop = () => {
      const blob = new Blob(recordedChunks, { type: "video/webm" });
      const url = URL.createObjectURL(blob);
      const a = document.createElement("a");
      a.href = url;
      a.download = "mirrored_video.webm"; // 下载的文件名可以根据需要修改
      document.body.appendChild(a);
      a.click();
      window.URL.revokeObjectURL(url);
      document.body.removeChild(a);
    };
    mediaRecorder.start();
  };

  const stopRecording = () => {
    mediaRecorder.stop();
  };

  const drawFrame = () => {
    canvasContext.clearRect(0, 0, canvasElement.width, canvasElement.height);

    // 绘制圆角矩形
    const cornerRadius = 30;
    canvasContext.save();
    canvasContext.beginPath();
    canvasContext.moveTo(cornerRadius, 0);
    canvasContext.arcTo(
      canvasElement.width,
      0,
      canvasElement.width,
      canvasElement.height,
      cornerRadius
    );
    canvasContext.arcTo(
      canvasElement.width,
      canvasElement.height,
      0,
      canvasElement.height,
      cornerRadius
    );
    canvasContext.arcTo(0, canvasElement.height, 0, 0, cornerRadius);
    canvasContext.arcTo(0, 0, canvasElement.width, 0, cornerRadius);
    canvasContext.closePath();
    canvasContext.clip();

    // 水平镜像翻转绘制
    canvasContext.scale(-1, 1);
    canvasContext.drawImage(
      videoElement,
      -canvasElement.width,
      0,
      canvasElement.width,
      canvasElement.height
    );
    canvasContext.restore();

    requestAnimationFrame(drawFrame);
  };

  videoElement.onplay = () => {
    requestAnimationFrame(drawFrame);
  };

  // 添加一个按钮，点击开始录制视频
  // const startButton = document.createElement('button');
  // startButton.textContent = '开始录制';
  // startButton.onclick = startRecording;
  // document.body.appendChild(startButton);
  startRecording();

  // 添加一个按钮，点击停止录制视频
  const stopButton = document.createElement("button");
  stopButton.textContent = "停止录制";
  stopButton.onclick = stopRecording;
  document.body.appendChild(stopButton);
};
