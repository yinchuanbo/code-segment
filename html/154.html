使用联合类型定义多类型变量

function printAny(un: any) {
  console.log(un);
  console.log(typeof un);
}
function printUnion(un: string | number) {
  console.log(un);
  console.log(typeof un);
}

let un: string | number;

un = "Union Types";

printAny(un);
printUnion(un);

un = 1;

printAny(un);
printUnion(un);

let b_un = true;

printAny(b_un);
// printUnion(b_un); // error

在使用联合类型时，如果传递的参数类型不在其定义类型的范围内，则 tsc 编译器会报错。

此外，如果一个对象是联合类型，那么只能访问此联合类型的所有类型中所共有的成员。

使用对象联合类型测试访问共有成员的应用

interface ITypeA {
  funcA(): void;
  funcZ(): void;
}

interface ITypeB {
  funcB(): void;
  funcZ(): void;
}

class ClzTypeA implements ITypeA {
  funcA(): void {
    console.log("funcA");
  }
  funcZ(): void {
    console.log("funcZ-A");
  }
}

class ClzTypeB implements ITypeB {
  funcB(): void {
    console.log("funcB");
  }
  funcZ(): void {
    console.log("funcZ-B");
  }
}

function getTypeAB(type: number): ClzTypeA | ClzTypeB {
  if (type === 1) {
    return new ClzTypeA();
  }
  if (type === 2) {
    return new ClzTypeB();
  }
  throw new Error(`Excepted ClzTypeA or ClzTypeB, got ${type}`);
}

let typeA = getTypeAB(1);
typeA.funcZ();

// typeA.funcA(); error

let typeB = getTypeAB(2);
typeB.funcZ();

// typeB.funcB(); error
