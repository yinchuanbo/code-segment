TS 交叉类型

2024-06-23 17:23:44 · YinHao

交叉类型：通过与（&）操作符将需要交叉的类型串联在一起。

interface IUser {
  name: string;
  gender: string;
  age: number;
}

interface IGrade {
  grade: number;
  class: number;
}

interface IFamily {
  mother: string;
  father: string;
  telphone: number;
}
// 交叉类型
type stuinfo = IUser & IGrade & IFamily;

class Student {
  stu: stuinfo;
  constructor(stu: stuinfo) {
    this.stu = stu;
  }
  print(): void {
    console.log(this.stu);
  }
  printName(): void {
    console.log(this.stu.name);
  }
}

const si: stuinfo = {
  name: "cici",
  gender: "girl",
  age: 8,
  grade: 3,
  class: 7,
  mother: "tina",
  father: "king",
  telphone: 88888888,
};

let ss = new Student(si);

ss.print();
ss.printName();
