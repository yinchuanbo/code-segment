  width: 8px;
  height: 8px;
}

/*定义滚动条轨道 内阴影+圆角*/

  background-color: #fafafa;
}

/*定义滑块 内阴影+圆角*/
::-webkit-scrollbar-thumb {
  border-radius: 10px;
  background: rgb(191, 191, 191);
}
