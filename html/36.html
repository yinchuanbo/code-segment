数组去重

  1. 方式一
const arr = [
  "justin1go",
  "justin2go",
  "justin2go",
  "justin3go",
  "justin3go",
  "justin3go",
];
const uniqueArr = Array.from(new Set(arr));
// const uniqueArr = [...new Set(arr)];
  1. 方式二
const arr = [
  "justin1go",
  "justin2go",
  "justin2go",
  "justin3go",
  "justin3go",
  "justin3go",
];
const uniqueArr = [];
arr.forEach((item) => {
  // 或者!uniqueArr.includes(item)
  if (uniqueArr.indexOf(item) === -1) {
    uniqueArr.push(item);
  }
});

const arr = [
  "justin1go",
  "justin2go",
  "justin2go",
  "justin3go",
  "justin3go",
  "justin3go",
];
const uniqueArr = arr.filter((item, index) => {
  return arr.indexOf(item, 0) === index;
});

const arr = [
  "justin1go",
  "justin2go",
  "justin2go",
  "justin3go",
  "justin3go",
  "justin3go",
];
const uniqueArr = arr.reduce(
  (prev, cur) => (prev.includes(cur) ? prev : [...prev, cur]),
  []
);

console.log(arr.at(-1)); // 倒数第一个值
console.log(arr.at(-2)); // 倒数第二个值
console.log(arr.at(0)); // 正数第一个
console.log(arr.at(1)); // 正数第二个

交换两个值

let a = 1;
let b = 2;
[a, b] = [b, a];

const arr = ["justin1go", "justin2go", "justin3go"];
[arr[0], arr[2]] = [arr[2], arr[0]];

位运算

  • 使用 & 运算符判断一个数的奇偶
// 偶数 & 1 = 0
// 奇数 & 1 = 1
console.log(2 & 1); // 0
console.log(3 & 1); // 1
  • 使用 ~、>>、<<、>>>、| 来取整
console.log(~~6.83); // 6
console.log(6.83 >> 0); // 6
console.log(6.83 << 0); // 6
console.log(6.83 | 0); // 6
// >>>不可对负数取整
console.log(6.83 >>> 0); // 6
  • 使用 ^ 来完成值交换
var a = 5;
var b = 8;
a ^= b;
b ^= a;
a ^= b;
console.log(a); // 8
console.log(b); // 5
  • 使用 &、>>、| 来完成 rgb 值和 16 进制颜色值之间的转换
/**
 * 16进制颜色值转RGB
 * @param  {String} hex 16进制颜色字符串
 * @return {String}     RGB颜色字符串
 */
function hexToRGB(hex) {
  var hexx = hex.replace("#", "0x");
  var r = hexx >> 16;
  var g = (hexx >> 8) & 0xff;
  var b = hexx & 0xff;
  return `rgb(${r}, ${g}, ${b})`;
}

/**
 * RGB颜色转16进制颜色
 * @param  {String} rgb RGB进制颜色字符串
 * @return {String}     16进制颜色字符串
 */
function RGBToHex(rgb) {
  var rgbArr = rgb.split(/[^\d]+/);
  var color = (rgbArr[1] << 16) | (rgbArr[2] << 8) | rgbArr[3];
  return "#" + color.toString(16);
}
// -------------------------------------------------
hexToRGB("#ffffff"); // 'rgb(255,255,255)'
RGBToHex("rgb(255,255,255)"); // '#ffffff'
