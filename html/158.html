2024-06-23 21:30:25 · YinHao

在类中使用可选属性实现加法运算的应用

class Add_OP {
  x: number;
  y?: number;
  z: number;
  constructor(x: number, y?: number) {
    this.x = x;
    this.y = y || 0;
    this.z = this.x + this.y;
  }
}

let c1 = new Add_OP(1);
console.log(c1.z);

let c2 = new Add_OP(1, 2);
console.log(c2.z);

let clun = new Add_OP(1, undefined);
console.log(clun.z);
