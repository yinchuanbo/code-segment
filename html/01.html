2024-06-03 21:52:01 · YinHao

function typeOf(obj) {
  var _toString = Object.prototype.toString; // 引用 Object 的原型方法 toString()
  var _type = {
    undefined: "undefined",
    number: "number",
    boolean: "boolean",
    string: "string",
    "[object Function]": "function",
    "[object RegExp]": "regexp",
    "[object Array]": "array",
    "[object Date]": "date",
    "[object Error]": "error",
    // ... 这里可以继续扩展
  };
  return (
    _type[typeof obj] || _type[_toString.call(obj)] || (obj ? "object" : "null")
  );
}

// 使用
var _abs = Math.abs;
console.log(typeOf(_abs)); // "function"
